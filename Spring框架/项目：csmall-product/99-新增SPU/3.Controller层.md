/**  
 * SPU的控制器类  
 */  
@Slf4j  
@RestController  
@RequestMapping("/spu")  
@Api(tags = "7.SPU管理模块")  
@Validated  
public class SpuController {  
  
    @Autowired  
    private SpuService spuService;  
  
    // http://localhost:8090/spu/add-new?name=test001&description=hahahaha&sort=100  
    // http://localhost:8090/spu/add-new    @PostMapping("/add-new")  
    @ApiOperation("新增SPU")  
    @ApiOperationSupport(order = 100)  
    public JsonResult<Void> addNew(@Valid SpuAddNewDTO spuAddNewDTO) {  
        log.debug("开始处理【添加SPU】的请求，参数：{}", spuAddNewDTO);  
        spuService.addNew(spuAddNewDTO);  
        return JsonResult.success();  
    }  
  
}
