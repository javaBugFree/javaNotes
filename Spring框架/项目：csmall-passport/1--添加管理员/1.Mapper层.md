
AdminMapper接口：
```java
/**  
 * 处理管理员数据的Mapper接口  
 */  
public interface AdminMapper {  
  
    /**  
     * 插入数据  
     *  
     * @param admin 要插入的数据  
     * @return 插入的行数  
     */  
    int insert(Admin admin);  
  
    /**  
     * 根据用户名统计数量  
     *  
     * @param username 用户名  
     * @return 统计的数量  
     */  
    int countByUsername(String username);  
  
    /**  
     * 根据手机号码统计数量  
     *  
     * @param phone 手机号码  
     * @return 统计的数量  
     */  
    int countByPhone(String phone);  
  
    /**  
     * 根据电子邮箱统计数量  
     *  
     * @param email 电子邮箱  
     * @return 统计的数量  
     */  
    int countByEmail(String email);  
    
}
```
